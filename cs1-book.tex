\newcommand{\includeprelab}[1]{
	\chapter{#1}
	
	{\huge Prelab #1}	
	
	\begin{flushright}
	Your Name \underline{\hspace{2.5in}}
	\end{flushright}
	\noindent\makebox[\linewidth]{\rule{\paperwidth}{0.4pt}}

    \input{lab#1/lab#1Prelab.tex}
    \newpage
}  
	
\newcommand{\includelab}[1]{
	\section{In-Class Lab #1}	
    
	\begin{flushright}
	Your Name \underline{\hspace{2.5in}}
	
	Your Partner's Name(s) \underline{\hspace{2.5in}}
	
	 \underline{\hspace{2.5in}}
	\end{flushright}
    \input{lab#1/lab#1Lab.tex}
    \newpage
}

	\section{Postlab #1 Instructions}	
    \input{lab#1/lab#1Postlab.tex}
}
